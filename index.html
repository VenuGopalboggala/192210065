const express = require('express'); const axios = require('axios'); const cors = require('cors'); const app = express(); app.use(cors()); const BASE_URL = 'http://20.244.56.144/evaluation-service'; const TOKEN = 'Bearer YOUR_ACCESS_TOKEN_HERE'; // Replace this with your actual access token // Utility to fetch all users const getAllUsers = async () => { const response = await axios.get(`${BASE_URL}/users`, { headers: { Authorization: TOKEN } }); return response.data.users; }; // Utility to fetch posts for a user const getUserPosts = async (userId) => { const response = await axios.get(`${BASE_URL}/users/${userId}/posts`, { headers: { Authorization: TOKEN } }); return response.data.posts; }; // Utility to fetch comments for a post const getPostComments = async (postId) => { const response = await axios.get(`${BASE_URL}/posts/${postId}/comments`, { headers: { Authorization: TOKEN } }); return response.data.comments; }; // GET /users â†’ Top 5 users with most comments across all their posts app.get('/users', async (req, res) => { try { const users = await getAllUsers(); const userCommentCounts = []; for (const [userId, userName] of Object.entries(users)) { const posts = await getUserPosts(userId); let totalComments = 0; for (const post of posts) { const comments = await getPostComments(post.id); totalComments += comments.length; } userCommentCounts.push({ userId, userName, totalComments }); } const topUsers = userCommentCounts .sort((a, b) => b.totalComments - a.totalComments) .slice(0, 5); res.json(topUsers); } catch (err) { res.status(500).json({ error: 'Error fetching top users.' }); } }); // GET /posts/type-popular?type=popular|latest app.get('/posts/type-popular', async (req, res) => { const type = req.query.type; try { const users = await getAllUsers(); let allPosts = []; for (const userId of Object.keys(users)) { const posts = await getUserPosts(userId); allPosts = allPosts.concat(posts); } if (type === 'latest') { const sorted = allPosts.sort((a, b) => b.id - a.id).slice(0, 5); res.json({ posts: sorted }); } else if (type === 'popular') { let commentMap = {}; for (const post of allPosts) { const comments = await getPostComments(post.id); commentMap[post.id] = comments.length; } const maxCount = Math.max(...Object.values(commentMap)); const topPosts = allPosts.filter(post => commentMap[post.id] === maxCount); res.json({ posts: topPosts }); } else { res.status(400).json({ error: 'Invalid type parameter' }); } } catch (err) { res.status(500).json({ error: 'Error fetching posts.' }); } }); app.listen(3001, () => { console.log('Backend server running on http://localhost:3001'); });
